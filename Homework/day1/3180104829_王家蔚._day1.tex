\subsection*{Newton Iteration}
\large INPUT: f, df, err=1, $x_0=2$ \\
\\
\large CODE \\
iter = 0;\\
err = 1;\\
x0 = 2.0;\\
x = x0;\\
format long;\\
while(err $>$ 1e-8 and iter < 20)\\
x0 = x;\\
x = x0 - df(x0)/f(x0);\\
err = norm(x - x0);\\
iter = iter +1;\\
fprintf('iter \%d: x = \%18.15f, f(x) = \%28.15f', iter, x, f(x))\\
end\\
